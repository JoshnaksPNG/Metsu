int foo = 0;
int bar = 0;

proc change_foo(int a)
{
	foo = a;
}

void change_bar(int a)
{
	if(a == 2)
	{
		return;
	}

	bar = a;
}

change_foo(1);
change_bar(2);

println(foo);
println(bar);
